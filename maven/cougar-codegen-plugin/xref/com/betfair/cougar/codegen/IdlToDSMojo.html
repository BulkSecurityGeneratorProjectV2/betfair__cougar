40  /**
41   * A plugin which is responsible for generating Cougar-based services. This encompasses a number
42   * of code- and file-generation steps, as well as validation. The intention is for this mojo to
43   * do everything needed, keeping the plugins/plugin section of a Cougar service's pom as simple as
44   * possible.
45   * <p>
46   * <h2>NOTE: idd dependencies</h2>
47   * IDDs can be read from the file system or as resources. The IDD is expected to be named after the
48   * service (see {@link #services} param), suffixed with {@code .xml} for the service and
49   * {@code -Extensions.xml} for the extensions definition. If you're using an IDD file then it should
50   *  be in {@code /src/main/resources}. Switch between the two modes using the {@link #iddAsResource}
51   *  flag.
52   * <p>
53   * A gotcha exists when accessing IDDs as resources.
54   * Since the IDDs are not required at run-time, it would make sense to include the relevant IDD
55   * project (jar) as a plugin dependency (ie. in {@code project/plugins/plugin/dependencies} as opposed to a
56   * project dependency of {@code project/dependencies}). You can do this <em>unless</em> your
57   * service is built as part of a larger project tree, in which multiple services are built. Maven
58   * resolves dependencies for the plugin once, so you can't have projectA relying on projectA-idd
59   * and project B relying on projectB-idd respectively - you end up with both projects relying on
60   * (say) projectA-idd. To work around this, you have to include the IDD as part of the project
61   * dependencies.
62   * <p>
63   * TODO If there's an easy way to fix this, we should do so (maven-savvy volunteers welcome)
64   *
65   * @goal process
66   *
67   * @phase generate-sources
68   * @requiresDependencyResolution
69   */
70  public class IdlToDSMojo extends AbstractMojo {

249     public void execute()
250         throws MojoExecutionException
251     {
252         getLog().info("Starting Cougar code generation");
253         if (isOffline()) {
254             getLog().warn("Maven in offline mode, plugin is NOT validating IDDs against schemas");
255         }
256         else {
257         	getLog().debug("Unbundling schemas for validation");
258         	catalogFile = unwrapSchemas();
259         }
260         initResourceLoader();
261         initResolver();	// needs the resource loader
262 
263         // load wsdl.xsl (as resource) and write (as file) to a working directory
264         prepWsdlXsl();
265 
266         // load xsd.xsl (as resource) and write (as file) to a working directory
267         prepXsdXsl();
268 
269         // load iddstripper.xsl (as resource) and write (as file) to a working directory
270         prepIddStripperXsl();
271 
272 
273         try {
274             getLog().debug("Starting IDL to Java");
275 
276             for (Service service : getServices() ) {
277                 processService(service);
278             }
279 
280             // this replaces the functionality of build-helper-maven-plugin
281             addSource();
282         }
283         catch (Exception e) {
284         	getLog().error(e);
285             throw new MojoExecutionException("Failed processing IDL: " + e, e);
286         }
287 
288         getLog().info("Completed Cougar code generation");
289     }

362     /**
363      * Find, open and parse the IDD implied by the specified service name. Reads either an explicit
364      * file or else a resource based on {@link #iddAsResource} flag.
365      * <p>
366      * The implied name is simply the service name + ".xml".
367      */
368     private Document parseIddFile(String serviceName) {
369 
370 		String iddFileName = serviceName + ".xml";
371 
372     	if (isIddAsResource()) {
373     		InputStream is = resourceLoader.getResourceAsStream(iddFileName);
374     		if (is == null) {
375     			throw new RuntimeException("Cannot open IDD resource named '" + iddFileName + "'");
376     		}
377     		return XmlUtil.parse(is, resolver);
378     	}
379     	else {
380     		File iddFile = new File( new File(getBaseDir(), RESOURCES_DIR), iddFileName);
381     		if (!iddFile.exists()) {
382     			throw new RuntimeException("Cannot open IDD file named '" + iddFileName + "'");
383     		}
384     		return XmlUtil.parse(iddFile, resolver);
385     	}
386 	}

400 	/**
401      * Find, open and parse the extensions xml file or null if it doesn't exist. Reads from
402      * file or resource based on {@link #iddAsResource} flag.
403      * <p>
404      * Name of extensions file should be ServiceName-Extensions.xml.
405      */
406     private Document parseExtensionFile(String serviceName) {
407 
408 		String extensionFileName = serviceName + "-Extensions.xml";
409 
410     	if (isIddAsResource()) {
411     		InputStream is = resourceLoader.getResourceAsStream(extensionFileName);
412     		if (is != null) {
413     			return XmlUtil.parse(is, resolver);
414     		}
415     		else {
416     			return null;
417     		}
418     	}
419     	else {
420     		File extensionsFile = new File( new File(getBaseDir(), RESOURCES_DIR), extensionFileName);
421     		if (extensionsFile.exists()) {
422         		return XmlUtil.parse(extensionsFile, resolver);
423     		}
424     		else {
425     			return null;
426     		}
427     	}

430 	/**
431      * The original concept of the IDLReader (other devs) has gone away a bit, so there could be
432      * some refactoring around this.
433      */
434 	private void generateJavaCode(Service service, Document iddDoc) throws Exception {
435 
436 		IDLReader reader = new IDLReader();
437 
438 		Document extensionDoc = parseExtensionFile(service.getServiceName());
439 
440 		String packageName = derivePackageName(service, iddDoc);
441 
442         reader.init(iddDoc, extensionDoc, service.getServiceName(), packageName, getBaseDir(),
443         				generatedSourceDir, getLog(), service.getOutputDir(), isClient(), isServer());
444 
445         runMerge(reader);
446 
447         // also create the stripped down, combined version of the IDD doc
448         getLog().debug("Generating combined IDD sans comments...");
449         Document combinedIDDDoc = parseIddFromString(reader.serialize());
450         // WARNING: this absolutely has to be run after a call to reader.runMerge (called by runMerge above) as otherwise the version will be null...
451         generateExposedIDD(combinedIDDDoc, reader.getInterfaceName(), reader.getInterfaceMajorMinorVersion());
452 
453         // generate WSDL/XSD
454         getLog().debug("Generating wsdl...");
455         generateWsdl(iddDoc, reader.getInterfaceName(), reader.getInterfaceMajorMinorVersion());
456         getLog().debug("Generating xsd...");
457         generateXsd(iddDoc, reader.getInterfaceName(), reader.getInterfaceMajorMinorVersion());
458 	}

460     /**
461 	 * Package name comes from explicit plugin param (if set), else the namespace definition, else
462 	 * skream and die.
463 	 * <p>
464 	 * Having the plugin override allows backwards compatibility as well as being useful for
465 	 * fiddling and tweaking.
466 	 */
467 	private String derivePackageName(Service service, Document iddDoc) {
468 
469 		String packageName = service.getPackageName();
470 		if (packageName == null) {
471 			packageName = readNamespaceAttr(iddDoc);
472 			if (packageName == null) {
473 				throw new PluginException("Cannot find a package name "
474 								+ "(not specified in plugin and no namespace in IDD");
475 			}
476 		}
477 		return packageName;
478 	}

541     @SuppressWarnings("unchecked")
542 	private void runMerge(IDLReader reader) throws Exception {
543         Transformations transformations = getTransformations();
544 
545         // First let's mangle the document if need be.
546         if (transformations.getManglers() != null) {
547         	getLog().debug("mangling IDL using "+transformations.getManglers().size()+" manglers");
548         	for(DocumentMangler m : transformations.getManglers()) {
549                 getLog().debug(m.getName());
550            	 	reader.mangle(m);
551             }
552         }
553 
554         if (transformations.getPreValidations() != null) {
555         	getLog().debug("Pre validating IDL using "+transformations.getPreValidations().size()+" pre validations");
556             for(Validator v : transformations.getPreValidations()) {
557                 getLog().debug(v.getName());
558                 reader.validate(v);
559             }
560 
561         }
562 
563         for(Transformation t : transformations.getTransformations()) {
564             getLog().debug(t.toString());
565         }
566         reader.runMerge(transformations.getTransformations());
567 
568         reader.writeResult();
569

628     private XPathExpression initNamespaceAttrExpression() {
629 
630 		XPathFactory xfactory = XPathFactory.newInstance();
631 		XPath xpath = xfactory.newXPath();
632 		try {
633 			return xpath.compile("/interface/@namespace");
634 		} catch (XPathExpressionException e) {
635 			throw new PluginException("Error compiling namespace XPath expression: " + e, e);
636 		}
637     }

639     /**
640      * Retrieve 'namespace' attr of interface definition or null if not found
641      */
642     private String readNamespaceAttr(Document doc) {
643 
644     	// lazy loading is mostly pointless but it keeps things together
645     	if (namespaceExpr == null) {
646     		namespaceExpr = initNamespaceAttrExpression();
647     	}
648 
649 		String s;
650 		try {
651 			s = namespaceExpr.evaluate(doc);
652 		} catch (XPathExpressionException e) {
653 			throw new PluginException("Error evaluating namespace XPath expression: " + e, e);
654 		}
655 		// xpath returns an empty string if not found, null is cleaner for callers
656 		return (s == null || s.length() == 0) ? null : s;
657     }
