17  package com.betfair.cougar.codegen;
18  
19  import java.util.Stack;
20  
21  import org.w3c.dom.Element;
22  import org.w3c.dom.Node;
23  
24  public class DomUtils {
25      public static String getNameBasedXPath(Node n, boolean includeCurrent) {
26  		Node parent = null;
27  		Stack<Node> hierarchy = new Stack<Node>();
28  
29  		if (includeCurrent) {
30  			hierarchy.push(n);
31  		}
32  		parent = n.getParentNode();
33  		while (null != parent && parent.getNodeType() != Node.DOCUMENT_NODE) {
34  			hierarchy.push(parent);
35  			parent = parent.getParentNode();
36  		}
37  
38  		// construct xpath
39  		StringBuilder sb = new StringBuilder();
40  		Node node = null;
41  		while (!hierarchy.isEmpty() && null != (node = hierarchy.pop())) {
42  			// only consider elements
43  			if (node.getNodeType() == Node.ELEMENT_NODE) {
44  				Element e = (Element) node;
45  
46  				// Bung in a slash and the element name
47  				sb.append("/");
48  				sb.append(node.getLocalName());
49  
50  				if (node.hasAttributes()) {
51  					// see if the element has a name attribute
52  					if (e.hasAttribute("name")) {
53  						// name attribute found - use that
54  						sb.append("[@name='").append(e.getAttribute("name")).append("']");
55  					}
56  				}
57  			}
58  		}
59  
60  		// return buffer
61  		return sb.toString();
62  	}
63  }
