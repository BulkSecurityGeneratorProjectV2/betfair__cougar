65      private Node getRootNode(NodeModel model) {
66          NodeList list = model.getNode().getChildNodes();
67          for (int i=0; i<list.getLength(); i++) {
68              Node n = list.item(i);
69              if (n instanceof Element) {
70                  return n;
71              }
72          }
73          throw new IllegalStateException("Can't find root node!");
74      }

137     public void validate(Validator validator) throws Exception {
138         final XPathFactory factory = XPathFactory.newInstance();
139 
140         // We want the root node, not the document
141         final Node doc = getRootNode(dataModel);
142         final NodeList nodes = (NodeList) factory.newXPath().evaluate(validator.getXPath(), doc, XPathConstants.NODESET);
143         if (nodes.getLength() == 0 && validator.nodeMustExist()) {
144             throw new ValidationException("Node "+validator.getXPath()+" must exist according to "+validator.getName(), null);
145         }
146         for (int i = 0; i < nodes.getLength(); i++) {
147             validator.validate(nodes.item(i));
148         }

168     private void runTransformation(final XPath xp, final Node doc, List<Transformation> transformations)
169             throws Exception {
170         String serviceName = getAttribute(doc, "name");
171         interfaceName = serviceName;
172         final String pkg = (packageName!=null)?packageName.replaceAll("\\.","/"):serviceName.toLowerCase();
173         // Can't have dots in a package name, so replace with underscores
174         interfaceMajorMinorVersion = "v" + getAttribute(doc, "version").replaceAll("\\.", "_");
175         // Ensure that the interfaceMajorMinorVersion is only vMajor.Minor, not vMajor.Minor.Revision
176         if (interfaceMajorMinorVersion.matches("v\\d+_\\d+_\\d+")) {
177             interfaceMajorMinorVersion = interfaceMajorMinorVersion.substring(0, interfaceMajorMinorVersion.lastIndexOf("_"));
178         }
179         interfaceMajorVersion = interfaceMajorMinorVersion;
180         // Ensure that the interfaceMajorVersion is only vMajor, not vMajor.Minor
181         if (interfaceMajorVersion.matches("v\\d+_\\d+")) {
182             interfaceMajorVersion = interfaceMajorVersion.substring(0, interfaceMajorVersion.lastIndexOf("_"));
183         }
184 
185         // Prepare the directories. This must be done before the generation loop
186         // as otherwise the additive files overwrite each other (jaxb.index)
187         for (final Transformation trans : transformations) {
188             if (isActive(trans)) {
189                 prepareDirectory(pkg, interfaceMajorVersion, interfaceMajorMinorVersion, trans);
190             }
191         }
192 
193         for (final Transformation trans : transformations) {
194             log.debug(trans.toString());
195             log.debug(trans.getNodePath());
196             log.debug(trans.getAdditionalNodesParameter());
197 
198             if (isActive(trans)) {
199                 final XPathExpression expr = xp.compile(trans.getNodePath());
200                 final NodeList nodes = (NodeList) expr.evaluate(doc, XPathConstants.NODESET);
201                 // find the xpath to give to the template as a parameter
202                 int parameterColonIndex = trans.getAdditionalNodesParameter() != null ? trans.getAdditionalNodesParameter().indexOf(":") : -1;
203                 List<NodeModel> parameterNodes = null;
204                 String parameterName = null;
205                 if (parameterColonIndex > 0) {
206                     parameterName = trans.getAdditionalNodesParameter().substring(0, parameterColonIndex);
207                     final XPathExpression paramExpr = xp.compile(trans.getAdditionalNodesParameter().substring(parameterColonIndex+1));
208                     parameterNodes = new ArrayList<NodeModel>();
209                     NodeList parameterNodeList = (NodeList) paramExpr.evaluate(doc, XPathConstants.NODESET);
210                     for (int i=0; i<parameterNodeList.getLength(); i++) {
211                         parameterNodes.add(NodeModel.wrap(parameterNodeList.item(i)));
212                     }
213                 }
214 
215                 String dirName = getDirectory(pkg, interfaceMajorVersion, interfaceMajorMinorVersion, trans);
216 
217                 log.debug("" + nodes.getLength());
218                 for (int i = 0; i < nodes.getLength(); i++) {
219                     final Node node = nodes.item(i);
220                     String nodeName = getName(node, trans.isCompositeName());
221 
222                     //Determine if an excluder implementation has been specified,
223                     //and if it has, visit that node to allow the user implementation
224                     //to exclude it
225                     if (trans.excluder == null || !trans.excluder.exclude(xp, node)) {
226                         writeTemplate(trans, NodeModel.wrap(doc), NodeModel.wrap(node), parameterName, parameterNodes, pkg, serviceName, interfaceMajorVersion, interfaceMajorMinorVersion, nodeName, dirName);
227                     }
228                 }
229             } else {
230                 log.debug(trans.toString() +" is inactive for this run - client: "+client
231                         + ", trans.isServer(): "+trans.isServer()
232                         + ", trans.isClient(): "+trans.isClient());
233             }
234         }
235     }

241     private void writeTemplate(final Transformation trans, final NodeModel topLevelDoc, final NodeModel data, final String paramName, final List<NodeModel> paramNodes, final String pkg, final String serviceName, final String majorVersion, final String majorMinorVersion, String name, String dirName)
242             throws Exception {
243         log.debug(trans.toString());
244         String nsVersion = majorVersion;
245         String namespace = "http://www.betfair.com/servicetypes/"+nsVersion+"/"+serviceName+"/";
246         final Map<String, Object> root = new HashMap<String, Object>();
247         root.put("interface", topLevelDoc);
248         root.put("doc", data);
249         root.put("package", pkg.replaceAll("/", "\\."));
250         root.put("majorVersion", majorVersion);
251         root.put("majorMinorVersion", majorMinorVersion);
252         root.put("name", name);
253         root.put("namespace", namespace);
254         if (paramName != null) {
255             if (paramNodes.size() == 1) {
256                 root.put(paramName, paramNodes.get(0));
257             }
258             else {
259                 root.put(paramName, paramNodes);
260             }
261         }
262 
263 
264         final File outputDirectory = new File(output, dirName);
265         name = name.trim();
266         name = capFirst(name);
267         final String fileName = trans.getFileName().replace("${name}", name);
268         final String className = fileName.replace(".java", "");
269         File outputFile = new File(outputDirectory, fileName);
270         FileWriter fw = new FileWriter(outputFile);
271         root.put("className", className);
272 
273         log.debug("Writing " + name + " using template " + trans.getTemplate() + " to file " + outputFile);
274         try {
275             // First thing is to deal with the service
276             final Template template = config.getTemplate(trans.getTemplate());
277             template.process(root, fw);
278         } finally {
279             fw.close();
280             //write the jaxb.index file
281             if(trans.isJaxb()) {
282                 outputFile = new File(outputDirectory, "jaxb.index");
283                 fw = new FileWriter(outputFile,true);
284                 fw.append(className).append("\n");
285                 fw.close();
286                 outputFile = new File(outputDirectory, "package-info.java");
287                 fw = new FileWriter(outputFile);
288                 fw.write("@javax.xml.bind.annotation.XmlAccessorOrder(javax.xml.bind.annotation.XmlAccessOrder.UNDEFINED)\n");
289                 fw.write("@javax.xml.bind.annotation.XmlSchema(namespace=\""+namespace+"\",elementFormDefault=javax.xml.bind.annotation.XmlNsForm.QUALIFIED)\n");
290                 fw.write("package "+dirName.replaceAll("/", ".")+";");
291                 fw.close();
292             }
293         }
294     }

296     private String capFirst(String name) {
297         final char firstChar = name.charAt(0);
298         if(!Character.isUpperCase(firstChar)) {
299             name = Character.toUpperCase(firstChar) + name.substring(1);
300         }
301         return name;
302     }

312     private String getName(final Node node, final boolean compositeName) {
313         Node currentNode = node;
314 
315         String nodeName = getAttribute(currentNode, "name");
316         while (nodeName == null) {
317             currentNode = currentNode.getParentNode();
318             nodeName = getAttribute(currentNode, "name");
319         }
320         if (compositeName) {
321             // we need to go up again to get the second part of the composite name
322             currentNode = currentNode.getParentNode();
323             String nodeNameStart = getAttribute(currentNode, "name");
324             while (nodeNameStart == null) {
325                 currentNode = currentNode.getParentNode();
326                 nodeNameStart = getAttribute(currentNode, "name");
327             }
328             nodeName =
329                     Character.toLowerCase(nodeNameStart.charAt(0)) +
330                             nodeNameStart.substring(1) +
331                             Character.toUpperCase(nodeName.charAt(0)) +
332                             nodeName.substring(1);
333         }
334         return nodeName;
335 
336     }

338     private void prepareDirectory(final String pkg, final String majorVersion, final String majorMinorVersion,
339                                   final Transformation trans) throws IOException {
340         // Deal with the output file
341         String dirName = trans.getDirectory().replace("${package}", pkg);
342         dirName = dirName.replace("${majorVersion}", majorVersion);
343         dirName = dirName.replace("${majorMinorVersion}", majorMinorVersion);
344         final File outputDirectory = new File(output, dirName);
345         log.debug("Preparing directory "+outputDirectory.getCanonicalPath());
346         if (!outputDirectory.exists()) {

348             if(!success) {
349                 throw new IOException("Could not create directory " + outputDirectory);
350             }
351         }
352         File outputFile = new File(outputDirectory, "jaxb.index");
353         FileWriter fw = new FileWriter(outputFile);
354         fw.close();

366     /**
367      * Weave the extensions defined in the extensions doc into the target.
368      */
369     private void mergeExtensionsIntoDocument(Node target, Node extensions) throws Exception {
370         final XPathFactory factory = XPathFactory.newInstance();
371         final NodeList nodes = (NodeList) factory.newXPath().evaluate("//extensions", extensions, XPathConstants.NODESET);
372         for (int i = 0; i < nodes.getLength(); i++) {
373             final Node extensionNode = nodes.item(i);
374             String nameBasedXpath =  DomUtils.getNameBasedXPath(extensionNode, false);
375             log.debug("Processing extension node: " + nameBasedXpath);
376 
377             final NodeList targetNodes = (NodeList) factory.newXPath().evaluate(nameBasedXpath, target, XPathConstants.NODESET);
378             if (targetNodes.getLength() != 1) {
379                 throw new IllegalArgumentException("XPath "+nameBasedXpath+" not found in target");
380             }
381             Node targetNode = targetNodes.item(0);
382             Node importedNode = targetNode.getOwnerDocument().importNode(extensionNode, true);
383             targetNode.appendChild(importedNode);
384         }
385     }

387     /**
388      * Cycle through the target Node and remove any operations not defined in the extensions document.
389      */
390     private void removeUndefinedOperations(Node target, Node extensions) throws Exception {
391         final XPathFactory factory = XPathFactory.newInstance();
392         final NodeList nodes = (NodeList) factory.newXPath().evaluate("//operation", target, XPathConstants.NODESET);
393         for (int i = 0; i < nodes.getLength(); i++) {
394             final Node targetNode = nodes.item(i);
395             String nameBasedXpath = DomUtils.getNameBasedXPath(targetNode, true);
396             log.debug("Checking operation: " + nameBasedXpath);
397 
398             final NodeList targetNodes = (NodeList) factory.newXPath().evaluate(nameBasedXpath, extensions, XPathConstants.NODESET);
399             if (targetNodes.getLength() == 0) {
400                 // This operation is not defined in the extensions doc
401                 log.debug("Ignoring IDL defined operation: " + getAttribute(targetNode, "name"));
402                 targetNode.getParentNode().removeChild(targetNode);
403             }
404         }
405     }

408     public String serialize() throws Exception {
409 
410         TransformerFactory tfactory = TransformerFactory.newInstance();
411         Transformer serializer = tfactory.newTransformer();
412         //Setup indenting to "pretty print"
413         serializer.setOutputProperty(OutputKeys.INDENT, "yes");
414         serializer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "2");
415 
416         ByteArrayOutputStream out = new ByteArrayOutputStream();
417         serializer.transform(new DOMSource(dataModel.getNode()), new StreamResult(out));
418         return out.toString();
419     }
