30  /**
31   * remove all simple types defined (except for valid values ones) and
32   * replace all references to them with their result. 
33   */
34  public class SimpleTypeMangler extends AbstractMangler {
35  
36  	@Override
37  	public String getName() {
38  		return "Simple Type Mangler";
39  	}
40  
41  	@Override
42  	public void mangleDocument(Node doc) {
43  		try {
44  			// First thing to is to get all simple types defined in the IDL
45  			List<Node> simpleTypes = getChildrenWithName(getName(), doc, "simpleType");
46  			// a container for checking if a simple type is already defined 
47  			Set<String> definedSimpleTypes = new HashSet<String>();
48  			for (Node st: simpleTypes) {
49  				List<Node> children = getChildrenWithName(getName(), st, "validValues");
50  				if (children.isEmpty()) {
51  					String name = getAttribute(getName(), st, "name");
52  					String type = getAttribute(getName(), st, "type");
53  					
54  					// check for simpleType duplicates
55  					if(definedSimpleTypes.contains(name)){
56  						throw new ValidationException("The data type " + name + " is already defined", st);
57  					}else{
58  						definedSimpleTypes.add(name);
59  					}
60  					
61  					replaceSimpleType(doc, name, type);
62  					st.getParentNode().removeChild(st);
63  				}
64  			}
65  		} catch (ValidationException e) {
66  			throw new IllegalArgumentException("Unable to mangle document", e);
67  		}
68  
69  	}
70  	
71  	private void replaceSimpleType(Node doc, String simpleTypeName, String simpleTypePrimitive) throws ValidationException {
72  		replaceViaXPath(doc, simpleTypeName, simpleTypePrimitive, "//parameter");
73  		replaceViaXPath(doc, simpleTypeName, simpleTypePrimitive, "//simpleResponse");
74  		replaceViaXPath(doc, simpleTypeName, simpleTypePrimitive, "//response");
75  	}
76  	
77  	private void replaceViaXPath(Node doc, String simpleTypeName, String simpleTypePrimitive, String xPath)  throws ValidationException {
78  		// Need to zap through the entire document and replace anything that has a type
79  		// of the "name" parameter, and replace it's value with the "type" parameter.
80  		final XPathFactory factory = XPathFactory.newInstance();
81          NodeList nodes;
82          try {
83      		nodes = (NodeList) factory.newXPath().evaluate(xPath, doc, XPathConstants.NODESET);
84  		} catch (XPathExpressionException e) {
85  			throw new ValidationException("Unable to parse XPath "+xPath, doc, e);
86  		}
87          for (int i = 0; i < nodes.getLength(); i++) {
88          	Node n = nodes.item(i);
89          	Node attrNode = getAttributeNode(getName(), n, "type");
90          	String content = attrNode.getTextContent();
91          	if (content.equals(simpleTypeName)) {
92          		attrNode.setTextContent(simpleTypePrimitive);
93          		
94          	} else {
95          		// Need to check if it's a composite type and replace there.
96          		String[] composites = getComposites(content);
97          		boolean contentChanged = false;
98          		if (composites != null) {
99          			// It is a composite type - check the underlying data types
100         			for (String composite: composites) {
101         	        	if (composite.equals(simpleTypeName)) {
102         	        		content = content.replace(simpleTypeName, simpleTypePrimitive);
103         	        		contentChanged = true;
104         	        	}        				
105         			}
106         			if (contentChanged) {
107         				attrNode.setTextContent(content);
108         			}
109         		}
110         	}
111         }

113 	}
