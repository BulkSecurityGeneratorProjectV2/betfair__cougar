19  import javax.xml.xpath.XPathConstants;
20  import javax.xml.xpath.XPathExpressionException;
21  import javax.xml.xpath.XPathFactory;
22  
23  import org.w3c.dom.Attr;
24  import org.w3c.dom.Element;
25  import org.w3c.dom.NamedNodeMap;
26  import org.w3c.dom.Node;
27  import org.w3c.dom.NodeList;
28  
29  /**
30   * BSIDL 3.1 supports defining a response as both a "response" and a "simpleResponse". This mangler
31   * simply converts "response" elements to "simpleResponse" elements so that we can continue to
32   * work with the same old ftls. It throws an IllegalArgumentException if the idd specifies both - this is
33   * allowed by the xsd, but is complete nonsense.
34   * <p/>
35   * If at some point support for "simpleResponse" is dropped from BSIDL then this mangler can be removed and
36   * the ftls updated to check for "response" instead.
37   */
38  public class ResponseToSimpleResponseMangler extends AbstractMangler {
39  
40      @Override
41      public String getName() {
42          return "ResponseToSimpleResponseMangler";
43      }
44  
45      @Override
46      public void mangleDocument(Node doc) {
47  
48          final XPathFactory factory = XPathFactory.newInstance();
49  
50          try {
51              //Get all parameters nodes, and recurse through to find any that have a "response" element defined
52              NodeList nodes = (NodeList)factory.newXPath().evaluate("//parameters", doc, XPathConstants.NODESET);
53              if (nodes != null) {
54                  for (int i = 0; i < nodes.getLength(); i++) {
55                      Node parametersNode = nodes.item(i);
56                      Node toBeReplaced = getNodeToBeReplaced(parametersNode);
57                      if (toBeReplaced != null) {
58                          //There is no way to simply change the qualifiedName or localName of the element, so we need to
59                          //go through all this palarva of making a copy and replacing it
60                          String qualifiedName = toBeReplaced.getPrefix() != null ? toBeReplaced.getPrefix() + ".simpleResponse" : "simpleResponse";
61                          Element replacementNode = parametersNode.getOwnerDocument().createElementNS(toBeReplaced.getNamespaceURI(), qualifiedName);
62                          //Copy all the child nodes (which surprisingly does not include attributes, even though they are nodes)
63                          NodeList replacementChildNodes = toBeReplaced.getChildNodes();
64                          if (replacementChildNodes != null) {
65                              for (int j = 0; j < replacementChildNodes.getLength(); j++) {
66                                  replacementNode.appendChild(replacementChildNodes.item(j).cloneNode(true));
67                              }
68                          }
69                          //Copy all the attributes
70                          NamedNodeMap attributes = toBeReplaced.getAttributes();
71                          if (attributes != null) {
72                              for (int j = 0; j < attributes.getLength(); j++) {
73                                  Attr attribute = (Attr)attributes.item(j);
74                                  //Strangely, it is possible to get a null attribute from this collection... check for it
75                                  if (attribute != null) {
76                                      replacementNode.setAttributeNode((Attr)attribute.cloneNode(true));
77                                  }
78                              }
79                          }
80                          //Replace the existing node with a new
81                          parametersNode.replaceChild(replacementNode, toBeReplaced);
82                      }
83                  }
84              }
85          } catch (XPathExpressionException e) {
86              throw new IllegalArgumentException("XPath failed to get parameters", e);
87          }
88      }
89  
90      //Check for a "response" element, which will need to be converted to a "simpleResponse" element
91      //Also ensures only one of the two has been defined
92      private Node getNodeToBeReplaced(Node parametersNode) {
93          Node toBeReplaced = null;
94          int responseCount = 0;
95          NodeList childNodes = parametersNode.getChildNodes();
96          if (childNodes != null) {
97              for (int j = 0; j < childNodes.getLength(); j++) {
98                  Node childNode = childNodes.item(j);
99                  String name=childNode.getLocalName();
100                 if ("simpleResponse".equals(name) || "response".equals(name)) {
101                     responseCount++;
102                     if("response".equals(name)){
103                         //This is the node that will need to be replaced with a copy with a localName of "simpleResponse"
104                         toBeReplaced=childNode;
105                     }
106                 }
107                 //If responseCount>1 This implies that both a simpleResponse and response have
108                 //been defined - this is allowed by the xsd, but makes no sense
109                 if (responseCount > 1) {
110                     throw new IllegalArgumentException("Only one of either simpleResponse or response should define the response type");
111                 }
112             }
113         }
114         return toBeReplaced;
115     }
