30  public class UnknownDataTypeValidator extends AbstractValidator{
31  	private Set<String> dataTypesDefined = new HashSet<String>();
32  	
33  	public UnknownDataTypeValidator() {
34  		// Add common data types
35  		dataTypesDefined.add("bool");
36  		dataTypesDefined.add("byte");
37  		dataTypesDefined.add("i32");
38  		dataTypesDefined.add("i64");
39  		dataTypesDefined.add("float");
40  		dataTypesDefined.add("double");
41  		dataTypesDefined.add("string");
42  		dataTypesDefined.add("dateTime");
43          dataTypesDefined.add("void");
44  	}
45  	@Override
46  	public boolean nodeMustExist() {
47  		return true;
48  	}
49  
50  	@Override
51  	public String getName() {
52  		return "Unknown Data Type Validator";
53  	}
54  
55  	@Override
56  	public String getXPath() {
57  		return "/interface";
58  	}
59  
60  	@Override
61  	public void validate(Node node) throws ValidationException {
62  		List<Node> types = getChildrenWithName(getName(), node, "dataType");
63  		for (Node dt: types) {
64  			String name = getAttribute(getName(), dt, "name");
65  			dataTypesDefined.add(name);
66  		}
67  		types = getChildrenWithName(getName(), node, "simpleType");
68  		for (Node dt: types) {
69  			String name = getAttribute(getName(), dt, "name");
70  			dataTypesDefined.add(name);
71  		}
72  		try {
73  			final XPathFactory factory = XPathFactory.newInstance();
74  	        NodeList nodes;
75  			nodes = (NodeList) factory.newXPath().evaluate("//parameter", node, XPathConstants.NODESET);
76  	        for (int i = 0; i < nodes.getLength(); i++) {
77  	        	Node n = nodes.item(i);
78  	        	String type = getAttribute(getName(), n, "type");
79  	        	if (!checkComposite(type, n)) {
80  	        		validateDataType(type, n);
81  	        	}
82  	        }
83  			nodes = (NodeList) factory.newXPath().evaluate("//simpleResponse", node, XPathConstants.NODESET);
84  	        for (int i = 0; i < nodes.getLength(); i++) {
85  	        	Node n = nodes.item(i);
86  	        	String type = getAttribute(getName(), n, "type");
87  	        	if (!checkComposite(type, n)) {
88  	        		validateDataType(type, n);
89  	        	}
90  	        }
91  		} catch (XPathExpressionException e) {
92  			throw new ValidationException("Unable to parse XPath //parameter", node, e);
93  		}
94  
95  	}
96  
97  	
98  	private void validateDataType(String type, Node n)  throws ValidationException {
99      	if (!dataTypesDefined.contains(type)) {
100     		throw new ValidationException("Data type "+type+" is not valid in this IDL", n);
101     	}
102 	}
103 	
104 	private boolean checkComposite(String type, Node n)  throws ValidationException {
105 		String[] composites = getComposites(type);
106 			for (String composite: composites) {
107 				validateDataType(composite, n);	
108 			}
109         return composites.length>0;
110 	}
