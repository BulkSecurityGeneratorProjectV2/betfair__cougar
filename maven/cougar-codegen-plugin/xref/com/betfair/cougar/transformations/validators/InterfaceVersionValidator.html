17  package com.betfair.cougar.transformations.validators;
18  
19  import com.betfair.cougar.codegen.ValidationException;
20  import org.w3c.dom.Node;
21  
22  import java.util.HashSet;
23  import java.util.Set;
24  
25  /**
26   * Requires that versions only be of the form x.y.
27   */
28  public class InterfaceVersionValidator extends AbstractValidator{
29  	private Set<String> namesUsed = new HashSet<String>();
30  
31  	@Override
32  	public boolean nodeMustExist() {
33  		return true;
34  	}
35  
36  	@Override
37  	public String getName() {
38  		return "Interface Version Validator";
39  	}
40  
41  	@Override
42  	public String getXPath() {
43  		return "/interface";
44  	}
45  
46  	@Override
47  	public void validate(Node node) throws ValidationException {
48  		// Check that the data type name starts with a upper case letter
49          String version = getAttribute(getName(), node, "version");
50          if (version == null || version.length() < 1) {
51              throw new ValidationException("Interface must have a version", node);
52          }
53          String[] split = version.split("\\.");
54          if (split.length!=2) {
55              throw new ValidationException("Interface version must be of the form x.y: "+version, node);
56          }
57          for (String s : split) {
58              try {
59                  int i = Integer.parseInt(s);
60                  if (i<0) {
61                      throw new ValidationException("All components of the interface version must be numeric and >=0: "+version, node);
62                  }
63              }
64              catch (NumberFormatException nfe) {
65                  throw new ValidationException("All components of the interface version must be numeric: "+version, node);//NOSONAR
66              }
67          }
68  	}
69  }
