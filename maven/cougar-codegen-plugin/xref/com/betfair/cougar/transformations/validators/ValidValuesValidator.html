17  package com.betfair.cougar.transformations.validators;
18  
19  import java.util.HashSet;
20  import java.util.List;
21  import java.util.Set;
22  
23  import org.w3c.dom.Node;
24  
25  import com.betfair.cougar.codegen.ValidationException;
26  
27  public class ValidValuesValidator extends AbstractValidator {
28      @Override
29      public boolean nodeMustExist() {
30          return false;
31      }
32  
33      @Override
34      public String getName() {
35          return "Valid Values Validator";
36      }
37  
38      @Override
39      public String getXPath() {
40          return "//validValues";
41      }
42  
43      @Override
44      public void validate(Node node) throws ValidationException {
45          // First thing is to find out if the valid values must have an ID
46          Node parent = node.getParentNode();
47          Node grandParent = parent.getParentNode();
48  
49          boolean needsIDs = false;
50          if (grandParent.getLocalName().equals("exceptionType")) {
51              // Only the first child of an exception type needs an ID
52              Node firstParam = getFirstChildWithName(grandParent, "parameter");
53              if (firstParam == null) {
54                  throw new ValidationException(getName() + " - Node does not have any parameter children defined", node);
55              }
56              if (firstParam.equals(parent)) {
57                  needsIDs = true;
58              }
59          }
60  
61          // Ensure that all enumerations are of type string.
62          String type = getAttribute(getName(), parent, "type");
63          if (!type.equals("string")) {
64              throw new ValidationException(getName() + " - Valid values is not of base type string.", node);
65          }
66  
67          // If it's a simpleType enumeration, it must start with an upper case letter
68          if (parent.getLocalName().equals("simpleType")) {
69              String name = getAttribute(getName(), parent, "name");
70              if (name == null || name.length() < 1) {
71                  throw new ValidationException("Data types must have a name", node);
72              } else if (Character.isLowerCase(name.charAt(0))) {
73                  throw new ValidationException("Simple type names must start with a capital letter", node);
74              }
75          }
76          Set<Integer> idsUsed = new HashSet<Integer>();
77          Set<String> namesUsed = new HashSet<String>();
78  
79          List<Node> values = getChildrenWithName(getName(), node, "value");
80          if (values.size() == 0) {
81              throw new ValidationException(getName() + " - Valid values list does not have any children defined", node);
82          }
83          for (Node val : values) {
84              Integer id = getAttributeAsInt(getName(), val, "id");
85  
86              if (needsIDs == (id == null)) {
87                  if (needsIDs) {
88                      throw new ValidationException(getName() + " - no ID defined for valid value", val);
89                  } else {
90                      throw new ValidationException(getName() + " - ID defined for valid value which does not require one", val);
91                  }
92              }
93  
94              if (id != null) {
95                  if (idsUsed.contains(id)) {
96                      throw new ValidationException(getName() + " - duplicate id: " + id, val);
97                  }
98                  idsUsed.add(id);
99              }
100             String vvName = getAttribute(getName(), val, "name");
101             if (vvName != null) {
102                 if (vvName.length() < 1) {
103                     throw new ValidationException("Valid values must have a name", node);
104                 } else if (Character.isLowerCase(vvName.charAt(0))) {
105                     throw new ValidationException("Valid value names must start with a capital letter", node);
106                 }
107                 if (namesUsed.contains(vvName)) {
108                     throw new ValidationException(getName() + " - duplicate name: " + vvName, val);
109                 }
110                 namesUsed.add(vvName);
111             } else {
112                 throw new ValidationException(getName() + " - no name defined for valid value", val);
113             }
114 
115 
116         }
117 
118     }
